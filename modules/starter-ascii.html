Step 1: String Multiplication Magic

In Python, you can multiply strings! This is the foundation of ASCII art.

# Multiply a string to repeat it!
print("*" * 10)      # 10 stars
print("-" * 20)      # 20 dashes
print("Ha" * 5)      # HaHaHaHaHa
print("=" * 30)

# Create a simple box
width = 20
print("+" + "-" * (width - 2) + "+")
print("|" + " " * (width - 2) + "|")
print("|" + " " * (width - 2) + "|")
print("+" + "-" * (width - 2) + "+")
Key concept: "*" * 5 creates "*****". This works with any string and any number!

Step 5: Text Banners

Turn any text into a fancy banner!

def make_banner(text, char="*", padding=2):
    """Create a text banner with a border"""
    text_len = len(text)
    width = text_len + (padding * 2) + 2  # +2 for border chars

    # Top border
    print(char * width)

    # Text with padding
    print(char + " " * padding + text + " " * padding + char)

    # Bottom border
    print(char * width)

# Try different styles!
make_banner("HELLO PYTHON!")

print()
make_banner("Welcome", char="#", padding=4)

print()
make_banner("Game Over", char="=", padding=3)
Reusability: Functions let you create banners with any text, border character, and padding - without rewriting code!

Step 6: Draw Boxes Around Text

Create professional-looking boxes for text!

def text_box(lines, padding=1):
    """Draw a box around multiple lines of text"""
    # Find the longest line
    max_len = max(len(line) for line in lines)
    width = max_len + (padding * 2)

    # Draw box
    print("+" + "-" * width + "+")
    for line in lines:
        # Pad each line to same width
        padded = line.ljust(max_len)
        print("|" + " " * padding + padded + " " * padding + "|")
    print("+" + "-" * width + "+")

# Single line
text_box(["Hello World!"])

print()

# Multiple lines
text_box([
    "Welcome to Python!",
    "Let's learn to code.",
    "It's fun!"
])

print()

# Menu style
text_box([
    "=== MENU ===",
    "1. Start Game",
    "2. Options",
    "3. Quit"
], padding=2)
The ljust() method: "Hi".ljust(10) pads the string with spaces to make it 10 characters: "Hi "

Step 7: Pattern Generator

Create a function that generates various patterns!

def draw_pattern(pattern_type, size=5, char="*"):
    """Draw different ASCII patterns"""

    if pattern_type == "triangle":
        for i in range(1, size + 1):
            print(char * i)

    elif pattern_type == "pyramid":
        for i in range(1, size + 1):
            spaces = " " * (size - i)
            stars = char * (2 * i - 1)
            print(spaces + stars)

    elif pattern_type == "diamond":
        # Top half
        for i in range(1, size + 1):
            spaces = " " * (size - i)
            stars = char * (2 * i - 1)
            print(spaces + stars)
        # Bottom half
        for i in range(size - 1, 0, -1):
            spaces = " " * (size - i)
            stars = char * (2 * i - 1)
            print(spaces + stars)

    elif pattern_type == "square":
        for i in range(size):
            print(char * size)

    elif pattern_type == "hollow_square":
        print(char * size)
        for i in range(size - 2):
            print(char + " " * (size - 2) + char)
        if size > 1:
            print(char * size)

# Demo all patterns!
patterns = ["triangle", "pyramid", "diamond", "square", "hollow_square"]

for p in patterns:
    print(f"\n{p.upper()}:")
    draw_pattern(p, size=5, char="*")

Step 8: Big Letter Art

Create block letters from characters!

# Block letter dictionary
letters = {
    'H': [
        "H   H",
        "H   H",
        "HHHHH",
        "H   H",
        "H   H"
    ],
    'I': [
        "IIIII",
        "  I  ",
        "  I  ",
        "  I  ",
        "IIIII"
    ],
    'P': [
        "PPPP ",
        "P   P",
        "PPPP ",
        "P    ",
        "P    "
    ],
    'Y': [
        "Y   Y",
        " Y Y ",
        "  Y  ",
        "  Y  ",
        "  Y  "
    ],
    ' ': [
        "     ",
        "     ",
        "     ",
        "     ",
        "     "
    ]
}

def print_big(text):
    """Print text in block letters"""
    text = text.upper()
    # Print row by row across all letters
    for row in range(5):
        line = ""
        for char in text:
            if char in letters:
                line += letters[char][row] + "  "
            else:
                line += "     " + "  "  # Unknown char
        print(line)

print_big("HI")
print()
print_big("PYPY")
How it works: Each letter is stored as 5 rows of text. We print row 0 of all letters, then row 1, etc. This creates the block effect!
