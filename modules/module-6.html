What is OOP?

Object-Oriented Programming organizes code around "objects" - things with data (attributes) and behaviors (methods).

# Define a class (blueprint)
class Dog:
    # __init__ runs when you create an object
    def __init__(self, name, age):
        self.name = name  # Attribute
        self.age = age    # Attribute
    
    # Method (function inside a class)
    def bark(self):
        print(f"{self.name} says Woof!")
    
    def birthday(self):
        self.age += 1
        print(f"{self.name} is now {self.age}!")

# Create objects (instances)
my_dog = Dog("Buddy", 3)
your_dog = Dog("Max", 5)

# Use attributes and methods
print(my_dog.name)    # Buddy
my_dog.bark()         # Buddy says Woof!
my_dog.birthday()     # Buddy is now 4!

The self Parameter

self refers to the current object. It's how methods access the object's data.

class BankAccount:
    def __init__(self, owner, balance=0):
        self.owner = owner
        self.balance = balance
    
    def deposit(self, amount):
        self.balance += amount
        return self.balance
    
    def withdraw(self, amount):
        if amount <= self.balance:
            self.balance -= amount
            return amount
        else:
            print("Insufficient funds!")
            return 0

# Using the class
account = BankAccount("Alice", 100)
account.deposit(50)      # balance: 150
account.withdraw(30)     # balance: 120
print(account.balance)   # 120

Inheritance

Create new classes based on existing ones. The child inherits from the parent.

# Parent class
class Animal:
    def __init__(self, name):
        self.name = name
    
    def speak(self):
        print(f"{self.name} makes a sound")

# Child classes inherit from Animal
class Dog(Animal):
    def speak(self):  # Override parent method
        print(f"{self.name} says Woof!")

class Cat(Animal):
    def speak(self):
        print(f"{self.name} says Meow!")
    
    def purr(self):   # New method only for Cat
        print(f"{self.name} purrs...")

# Using inheritance
dog = Dog("Buddy")
cat = Cat("Whiskers")

dog.speak()   # Buddy says Woof!
cat.speak()   # Whiskers says Meow!
cat.purr()    # Whiskers purrs...

Class vs Instance Attributes

class Player:
    # Class attribute (shared by all instances)
    game_name = "Adventure Quest"
    player_count = 0
    
    def __init__(self, username):
        # Instance attributes (unique to each object)
        self.username = username
        self.score = 0
        Player.player_count += 1  # Update class attribute

# All players share game_name
p1 = Player("Alice")
p2 = Player("Bob")

print(Player.game_name)    # Adventure Quest
print(Player.player_count) # 2
print(p1.username)         # Alice (instance-specific)

Special Methods (Dunder Methods)

class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y
    
    def __str__(self):
        # Called by print() and str()
        return f"Point({self.x}, {self.y})"
    
    def __add__(self, other):
        # Called by + operator
        return Point(self.x + other.x, self.y + other.y)
    
    def __eq__(self, other):
        # Called by == operator
        return self.x == other.x and self.y == other.y

p1 = Point(1, 2)
p2 = Point(3, 4)

print(p1)           # Point(1, 2)
p3 = p1 + p2        # Uses __add__
print(p3)           # Point(4, 6)
print(p1 == p2)     # False
