What Are Functions?

Functions are reusable blocks of code with a name. Define once, use many times!

# Define a function with def
def greet():
    print("Hello!")

# Call the function
greet()  # Output: Hello!
greet()  # Can call it again!

# Function with a parameter
def greet_person(name):
    print(f"Hello, {name}!")

greet_person("Alice")  # Hello, Alice!
greet_person("Bob")    # Hello, Bob!

Return Values

Functions can send back a result with return.

# Function that returns a value
def add(a, b):
    return a + b

result = add(5, 3)
print(result)  # 8

# Use returned value directly
print(add(10, 20))  # 30

# Multiple returns (tuple)
def get_name_age():
    return "Alice", 25

name, age = get_name_age()
print(name, age)  # Alice 25

Default & Keyword Arguments

# Default parameter values
def greet(name, greeting="Hello"):
    print(f"{greeting}, {name}!")

greet("Alice")           # Hello, Alice!
greet("Bob", "Hi")       # Hi, Bob!

# Keyword arguments (named)
def describe(name, age, city):
    print(f"{name}, {age}, from {city}")

# Can pass in any order with names
describe(age=30, city="NYC", name="Eve")

*args and **kwargs

Accept any number of arguments.

# *args: any number of positional args
def sum_all(*numbers):
    return sum(numbers)

print(sum_all(1, 2, 3))      # 6
print(sum_all(1, 2, 3, 4, 5)) # 15

# **kwargs: any number of keyword args
def print_info(**data):
    for key, value in data.items():
        print(f"{key}: {value}")

print_info(name="Alice", age=25, city="NYC")

Importing Modules

Modules are files with reusable code. Python has many built-in!

# Import entire module
import math
print(math.sqrt(16))  # 4.0
print(math.pi)        # 3.14159...

# Import specific items
from random import randint, choice
print(randint(1, 10))  # Random 1-10
print(choice(["a", "b", "c"]))

# Import with alias
import datetime as dt
print(dt.date.today())
