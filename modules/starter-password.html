# Sneak peek: Generate 5 secure passwords!
import random
import string

def generate_password(length=12):
    characters = string.ascii_letters + string.digits + string.punctuation

Step 1: Picking Random Items

The random module lets Python make random choices. random.choice() picks one item from a list.

import random

# random.choice() picks ONE random item
colors = ["red", "blue", "green", "yellow", "purple"]
print(f"Random color: {random.choice(colors)}")

# Works with strings too! (picks one character)
letters = "ABCDEFG"
print(f"Random letter: {random.choice(letters)}")

# Pick 5 random letters
print("\n5 random letters:")
for i in range(5):
    print(random.choice(letters), end=" ")
Key insight: A string is just a sequence of characters, so random.choice("ABC") picks one character: 'A', 'B', or 'C'.

Step 2: The String Module

Python's string module has pre-built character sets - no need to type them yourself!

import string

These are just strings - you can combine them with +!

Step 3: Building a Password

Let's pick random characters and join them together!

import random
import string

# All possible characters
all_chars = string.ascii_letters + string.digits

# Build a password character by character
password = ""
for i in range(8):
    random_char = random.choice(all_chars)
    password = password + random_char  # Add to password

print(f"Generated password: {password}")

# Shorter way using a list
password_chars = []
for i in range(8):
    password_chars.append(random.choice(all_chars))
password2 = "".join(password_chars)  # Join list into string

print(f"Another password: {password2}")
The join() method: "".join(["a", "b", "c"]) combines a list into a string: "abc". The separator (here "") goes between each item.

Step 4: Python Shortcut - List Comprehension

Python has a compact way to build lists in one line!

import random
import string

chars = string.ascii_letters + string.digits

# Long way (4 lines)
password_chars = []
for i in range(10):
    password_chars.append(random.choice(chars))
password1 = "".join(password_chars)

# Short way - list comprehension! (1 line)
password2 = "".join([random.choice(chars) for i in range(10)])

# Even shorter with generator (no brackets)
password3 = "".join(random.choice(chars) for _ in range(10))

print(f"Long way:  {password1}")
print(f"Short way: {password2}")
print(f"Shortest:  {password3}")
The underscore _: When you don't need the loop variable, use _ to show "I'm ignoring this value".

Step 6: Add More Options!

Real password generators let you choose what types of characters to include.

import random
import string

def generate_password(length=12,
                      use_uppercase=True,
                      use_lowercase=True,
                      use_digits=True,
                      use_symbols=True):
    """Generate a customizable password"""
    characters = ""

    if use_uppercase:
        characters += string.ascii_uppercase
    if use_lowercase:
        characters += string.ascii_lowercase
    if use_digits:
        characters += string.digits
    if use_symbols:
        characters += string.punctuation

    if not characters:
        return "Error: Select at least one character type!"

    return "".join(random.choice(characters) for _ in range(length))

# Different password styles
print("All options:", generate_password(16))
print("No symbols:", generate_password(16, use_symbols=False))
print("Only letters:", generate_password(16, use_digits=False, use_symbols=False))
print("Numbers only:", generate_password(8, use_uppercase=False, use_lowercase=False, use_symbols=False))
print("PIN code:", generate_password(4, use_uppercase=False, use_lowercase=False, use_symbols=False))

Step 7: Check Password Strength

Let's add a function to rate how strong a password is!

import string

def check_strength(password):
    """Rate password strength from 0-5"""
    score = 0
    feedback = []

    # Length check
    if len(password) >= 8:
        score += 1
    else:
        feedback.append("Too short (need 8+ chars)")

    if len(password) >= 12:
        score += 1

    # Character variety checks
    if any(c in string.ascii_lowercase for c in password):
        score += 1
    else:
        feedback.append("Add lowercase letters")

    if any(c in string.ascii_uppercase for c in password):
        score += 1
    else:
        feedback.append("Add uppercase letters")

    if any(c in string.digits for c in password):
        score += 1
    else:
        feedback.append("Add numbers")

    if any(c in string.punctuation for c in password):
        score += 1
    else:
        feedback.append("Add symbols")

    # Rating
    ratings = ["Very Weak", "Weak", "Fair", "Good", "Strong", "Very Strong", "Excellent"]

    return {
        "score": score,
        "rating": ratings[score],
        "feedback": feedback
    }

# Test some passwords
test_passwords = ["abc", "password123", "MyP@ssw0rd!", "X#9kL$mN2@pQ"]

for pwd in test_passwords:
    result = check_strength(pwd)
    print(f"\n'{pwd}'")
    print(f"  Score: {result['score']}/6 - {result['rating']}")
    if result['feedback']:
        print(f"  Tips: {', '.join(result['feedback'])}")
The any() function: any(condition for item in list) returns True if ANY item matches. Super useful for checking if something exists!

Step 8: The Complete Password Generator!

Let's put it all together into a mini password manager!

import random
import string

def generate_password(length=12, use_symbols=True):
    chars = string.ascii_letters + string.digits
    if use_symbols:
        chars += string.punctuation
    return "".join(random.choice(chars) for _ in range(length))

def check_strength(password):
    score = 0
    if len(password) >= 8: score += 1
    if len(password) >= 12: score += 1
    if any(c.islower() for c in password): score += 1
    if any(c.isupper() for c in password): score += 1
    if any(c.isdigit() for c in password): score += 1
    if any(c in string.punctuation for c in password): score += 1
    ratings = ["Very Weak", "Weak", "Fair", "Good", "Strong", "Very Strong", "Excellent"]
    return ratings[score]

# Generate passwords for different uses
print("=" * 50)
print("PASSWORD GENERATOR")
print("=" * 50)

uses = [
    ("Email account", 16, True),
    ("Banking", 20, True),
    ("Social media", 12, True),
    ("WiFi password", 16, False),
    ("PIN code", 6, False),
]

for use, length, symbols in uses:
    pwd = generate_password(length, symbols)
    strength = check_strength(pwd)
    print(f"\n{use}:")
    print(f"  Password: {pwd}")
    print(f"  Strength: {strength}")

print("\n" + "=" * 50)

import random

# Passphrase generator - easier to remember!
words = [
    "apple", "banana", "cherry", "dragon", "eagle",
    "forest", "galaxy", "harbor", "island", "jungle",
    "kitchen", "lemon", "mountain", "nebula", "ocean",
    "planet", "quantum", "river", "sunset", "thunder",
    "umbrella", "volcano", "window", "xylophone", "yellow"
]

def generate_passphrase(num_words=4, separator="-", capitalize=True):
    """Generate a memorable passphrase"""
    chosen_words = random.sample(words, num_words)

    if capitalize:
        chosen_words = [word.capitalize() for word in chosen_words]

    # Add a random number for extra security
    number = str(random.randint(10, 99))

    return separator.join(chosen_words) + separator + number

# Generate some passphrases
print("Memorable Passphrases:")
print("-" * 40)
for i in range(5):
    phrase = generate_passphrase()
    print(f"  {phrase}")

print("\nWith different separators:")
print(f"  Dashes: {generate_passphrase(separator='-')}")
print(f"  Dots:   {generate_passphrase(separator='.')}")
print(f"  None:   {generate_passphrase(separator='')}")
