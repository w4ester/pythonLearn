Files & Exceptions

PCEP 5.1-5.2Google Week 4

Reading Files

Python can read text files easily.

# Open and read entire file
with open("data.txt", "r") as file:
    content = file.read()
    print(content)

# Read line by line
with open("data.txt", "r") as file:
    for line in file:
        print(line.strip())  # strip() removes \n

# Read all lines into a list
with open("data.txt", "r") as file:
    lines = file.readlines()
    print(lines)  # ['line1\n', 'line2\n', ...]
üí° Why "with"? The with statement automatically closes the file when done, even if errors occur. Always use it!

Writing Files

# Write (creates or overwrites file)
with open("output.txt", "w") as file:
    file.write("Hello, World!\n")
    file.write("Second line\n")

# Append (adds to existing file)
with open("output.txt", "a") as file:
    file.write("This is added at the end\n")

# Write multiple lines
lines = ["Line 1", "Line 2", "Line 3"]
with open("output.txt", "w") as file:
    for line in lines:
        file.write(line + "\n")

Working with JSON

JSON is great for saving structured data.

import json

# Save data to JSON
data = {"name": "Alice", "scores": [90, 85, 88]}

with open("data.json", "w") as file:
    json.dump(data, file, indent=2)

# Load data from JSON
with open("data.json", "r") as file:
    loaded = json.load(file)
    print(loaded["name"])  # Alice

Handling Exceptions

Errors happen! Handle them gracefully with try/except.

# Basic try/except
try:
    number = int(input("Enter a number: "))
    result = 10 / number
    print(result)
except ValueError:
    print("That's not a valid number!")
except ZeroDivisionError:
    print("Can't divide by zero!")

# Catch multiple exceptions
try:
    risky_operation()
except (TypeError, ValueError) as e:
    print(f"Error: {e}")

# try/except/else/finally
try:
    file = open("data.txt")
except FileNotFoundError:
    print("File not found!")
else:
    print(file.read())  # Only if no error
    file.close()
finally:
    print("This always runs")
