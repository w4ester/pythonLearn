# AI in action: Minimax for Tic-Tac-Toe
def minimax(board, maximizing):
    if terminal(board):
        return utility(board)

    if maximizing:
        best = float('-inf')
        for action in actions(board):
            best = max(best, minimax(
                result(board, action), False
            ))
        return best
    else:
        best = float('inf')
        for action in actions(board):
            best = min(best, minimax(
                result(board, action), True
            ))
        return best
